Define function average for integers (array)
   Initialize sum to 0
   For each number in the array, add it to sum
   Return sum divided by the length of the array

Define function average for doubles (array)
   Initialize sum to 0
   For each number in the array, add it to sum
   Return sum divided by the length of the array

Define main function
   Create a scanner object for user input
   Create an array of doubles with length 10
   Prompt the user to enter ten numerical values, one at a time
   For each of the ten values:
      Check if the input is a double
         If not, display an error message and discard the invalid input
      Store the valid input in the array
   Calculate the average of the entered numbers using the average function for doubles
   Display the calculated average to the user

End


------------------------------------------


import java.util.*;

public class AverageArray {
    // Method to calculate the average from an array of ints
    public static int average(int[] array) {
        int sum = 0;
        for (int num : array) {
            sum += num;
        }
        return sum / array.length;
    }

    // Method to calculate the average of an array with doubles
    public static double average(double[] array) {
        double sum = 0;
        for (double num : array) {
            sum += num;
        }
        return sum / array.length;
    }
  
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        double[] numbers = new double[10];

        // Prompts the user to enter ten values
        System.out.println("Please enter ten numerical values, one at a time: ");
        for (int i = 0; i < 10; i++) {
            // Checks that input is valid as either int or double
            while (!scanner.hasNextDouble()) {
                System.out.println("Error: Invalid input. Please enter either a whole number or a decimal. ");
                scanner.next(); // Disregards the invalid input
            }
            numbers[i] = scanner.nextDouble();
        }

        // Displays result to the user
        System.out.println("\nAverage of the numbers: " + average(numbers));
    }
}
